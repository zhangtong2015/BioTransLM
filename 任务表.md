### 单元测试标准模板

```python
# tests/unit/test_sensory_gating.py
import pytest
import torch
from unittest.mock import Mock, patch

class TestSensoryGating:
    """感觉门控模块单元测试"""
    
    @pytest.fixture
    def config(self):
        return Mock(
            threshold=0.5,
            noise_reduction=0.3,
            hidden_dim=768
        )
    
    @pytest.fixture
    def sensory_gating(self, config):
        from biotranslm.input import SensoryGating
        return SensoryGating(config)
    
    @pytest.mark.unit
    def test_forward_shape(self, sensory_gating):
        """测试输出形状正确性"""
        batch_size, seq_len, dim = 2, 10, 768
        input_tensor = torch.randn(batch_size, seq_len, dim)
        
        result = sensory_gating.forward(input_tensor)
        
        assert result['gated_activations'].shape == (batch_size, seq_len, dim)
        assert 0 <= result['noise_level'] <= 1
    
    @pytest.mark.unit
    def test_noise_estimation(self, sensory_gating):
        """测试噪声估计算法"""
        # 纯噪声输入
        pure_noise = torch.randn(1, 5, 768) * 0.1
        noise_level = sensory_gating._estimate_noise(pure_noise)
        assert noise_level > 0.5
        
        # 干净信号
        clean_signal = torch.zeros(1, 5, 768)
        clean_signal[0, :, 0:10] = 1.0
```

### 性能测试标准

```python
# tests/e2e/test_performance.py
import pytest
import torch
import time
import psutil

class TestPerformance:
    """性能基准测试"""
    
    PERFORMANCE_THRESHOLDS = {
        'latency_per_sample_ms': 100,      # 单样本延迟 < 100ms
        'throughput_samples_sec': 50,      # 吞吐量 > 50样本/秒
        'memory_peak_gb': 8,               # 峰值内存 < 8GB
        'gpu_memory_peak_gb': 4,           # GPU峰值内存 < 4GB
    }
    
    @pytest.mark.performance
    def test_inference_latency(self, end_to_end_model, benchmark_dataset):
        """测试推理延迟"""
        model = end_to_end_model
        model.eval()
        
        latencies = []
        with torch.no_grad():
            for batch in benchmark_dataset:
                torch.cuda.synchronize() if torch.cuda.is_available() else None
                start = time.perf_counter()
                
                _ = model(batch['input_ids'])
                
                torch.cuda.synchronize() if torch.cuda.is_available() else None
                latency = (time.perf_counter() - start) * 1000  # ms
                latencies.append(latency)
        
        avg_latency = sum(latencies) / len(latencies)
        p99_latency = sorted(latencies)[int(len(latencies) * 0.99)]
        
        print(f"\n平均延迟: {avg_latency:.2f} ms")
        print(f"P99延迟: {p99_latency:.2f} ms")
        
        assert avg_latency < self.PERFORMANCE_THRESHOLDS['latency_per_sample_ms']
    
    @pytest.mark.performance
    def test_memory_usage(self, end_to_end_model, benchmark_dataset):
        """测试内存使用"""
        model = end_to_end_model
        model.eval()
        
        process = psutil.Process()
        initial_memory = process.memory_info().rss / 1024**3  # GB
        
        if torch.cuda.is_available():
            initial_gpu_memory = torch.cuda.memory_allocated() / 1024**3
            torch.cuda.reset_peak_memory_stats()
        
        with torch.no_grad():
            for i, batch in enumerate(benchmark_dataset):
                _ = model(batch['input_ids'])
                if i >= 10:  # 预热后测量
                    break
        
        peak_memory = process.memory_info().rss / 1024**3
        memory_increase = peak_memory - initial_memory
        
        print(f"\n内存增量: {memory_increase:.2f} GB")
        assert memory_increase < self.PERFORMANCE_THRESHOLDS['memory_peak_gb']
        
        if torch.cuda.is_available():
            peak_gpu_memory = torch.cuda.max_memory_allocated() / 1024**3
            print(f"GPU峰值内存: {peak_gpu_memory:.2f} GB")
            assert peak_gpu_memory < self.PERFORMANCE_THRESHOLDS['gpu_memory_peak_gb']
    
    @pytest.mark.performance
    def test_sparsity_efficiency(self, end_to_end_model, benchmark_dataset):
        """验证稀疏计算效率"""
        model = end_to_end_model
        
        # 测量稀疏模式
        activation_sparsities = []
        
        def hook(module, input, output):
            if isinstance(output, torch.Tensor):
                sparsity = (output == 0).float().mean().item()
                activation_sparsities.append(sparsity)
        
        # 注册钩子到主要层
        hooks = []
        for name, module in model.named_modules():
            if 'htm' in name or 'snn' in name or 'sparse' in name:
                hooks.append(module.register_forward_hook(hook))
        
        with torch.no_grad():
            for batch in benchmark_dataset:
                _ = model(batch['input_ids'])
                break
        
        for h in hooks:
            h.remove()
        
        avg_sparsity = sum(activation_sparsities) / len(activation_sparsities)
        print(f"\n平均激活稀疏度: {avg_sparsity:.2%}")
        assert avg_sparsity > 0.7  # >70%稀疏度目标
```

### 常见实现陷阱与规避

| 陷阱描述            | 规避策略                    | 检查点                        |
| --------------- | ----------------------- | -------------------------- |
| HTM微型柱数量过多导致OOM | 初始使用2048列而非4096列        | `n_columns <= 2048`        |
| 稀疏张量操作效率低下      | 使用`torch-sparse`库而非原生稀疏 | 使用`SparseTensor`           |
| STDP计算复杂度高      | 批处理优化 + 仅考虑窗口内脉冲        | `attention_span <= 128`    |
| 模块间通信数据拷贝过多     | 原地修改 + 共享内存队列           | 验证无冗余拷贝                    |
| FAISS索引搜索瓶颈     | HNSW索引 + 批量搜索           | `index.is_trained`         |
| 工作记忆衰减溢出        | sigmoid约束激活范围           | 激活∈\[0,1]                  |
| System2推理无限循环   | max\_depth硬限制 + 步数计数    | depth <= config.max\_depth |
